Get Started in Minutes

global
    log /dev/log    local0
    maxconn 2000

defaults
    log global
    mode http
    timeout connect 5000ms
    timeout client 50000ms
    timeout server 50000ms

frontend http_front
    bind *:80
    default_backend http_back

backend http_back
    balance roundrobin
    server web01 192.168.1.1:80 check
    server web02 192.168.1.2:80 check
